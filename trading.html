'gold' : symbol === 'BTCUSD' ? 'btc' : 'oil')); if (btn) btn.classList.add('active'); // reset demo candles for new asset candles = []; for (let i = 0; i < 15; i++) { let open = price + (Math.random() * 40 - 20); candles.push({ o: open, h: open + 25, l: open - 25, c: open + (Math.random() * 30 - 15) }); } initChart(); }; window.ex = function(action) { const symbol = document.getElementById('c-as').value; const price = parseFloat(document.getElementById('e-pr').value) || 0; const size = parseFloat(document.getElementById('e-sz').value) || 0; const feePct = parseFloat(document.getElementById('fee-pct').value) || 0; const taxPct = parseFloat(document.getElementById('tax-pct').value) || 0; const raw = price * size; const fees = raw * (feePct / 100); const afterFees = raw - fees; const tax = afterFees * (taxPct / 100); const adjusted = afterFees - tax; const now = new Date().toLocaleTimeString(); window.trades.unshift({ id: window.tradeId++, time: now, symbol, action, price, size, raw, fees, taxPct, adjusted }); // nudge last candle on trade if (candles.length) { const last = candles[candles.length - 1]; last.c = price; last.h = Math.max(last.h, price); last.l = Math.min(last.l, price); } initChart(); window.renderTables(); }; window.renderTables = function() { const tape = document.getElementById('t-body'); const ledger = document.getElementById('l-body'); tape.innerHTML = window.trades.map(t => `${t.time}${t.symbol}${t.action}${t.price.toFixed(2)}${t.size}` ).join('') || 'No trades yet'; ledger.innerHTML = window.trades.map(t => `#${t.id}${t.symbol}${t.action}$${t.raw.toFixed(2)}$${t.fees.toFixed(2)}${t.taxPct}%$${t.adjusted.toFixed(2)}` ).join('') || 'No ledger entries'; }; function initChart() { const canvas = document.getElementById('liveChart'); if (!canvas) return; const ctx = canvas.getContext('2d'); const w = canvas.width, h = canvas.height; ctx.fillStyle = '#090a0d'; ctx.fillRect(0, 0, w, h); if (!candles.length) return; const lows = candles.map(c => c.l); const highs = candles.map(c => c.h); const min = Math.min(...lows), max = Math.max(...highs); const pad = 20; const xStep = (w - pad * 2) / candles.length; candles.forEach((c, i) => { const x = pad + i * xStep + xStep / 2; const y = v => pad + (h - pad * 2) * (1 - (v - min) / (max - min || 1)); const up = c.c >= c.o; ctx.strokeStyle = ctx.fillStyle = up ? '#1c752b' : '#ce380f'; ctx.beginPath(); ctx.moveTo(x, y(c.h)); ctx.lineTo(x, y(c.l)); ctx.stroke(); const top = y(Math.max(c.o, c.c)); const bot = y(Math.min(c.o, c.c)); ctx.fillRect(x - 4, top, 8, Math.max(2, bot - top)); }); }
